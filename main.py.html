#creates patient class that has setters and getter, book or cancel appointment
class Patient:
    def __init__(self,patientName, patientID,phoneNumber):
        self.patientName = patientName
        self.patientID = patientID
        self.phoneNumber = phoneNumber
        self.appointments = []

    def setPatientName(self,patientName):
        self.patientName = patientName

    def getPatientName(self):
        return self.patientName

    def setPatientID(self,patientID):
        self.patientID = patientID

    def getPatientID(self):
        return self.patientID

    def setPhoneNumber(self,phoneNumber):
        self.phoneNumber = phoneNumber

    def getPhoneNumber(self):
        return self.phoneNumber

    def bookAppointment(self,appointment):
        self.appointments.append(appointment)

    def cancelAppointment(self, appointment):
        self.appointments.remove(appointment)

    def getAppointment(self):
        return self.appointments

#Appointemnt class that accesses objects created from different classes, has setters and getters , string function to display appointment information
class Appointment:
    def __init__(self,patient:Patient,date,dentist:Dentist, service:DentalService):
        self.patient = patient
        self.date = date
        self.dentist = dentist
        self.service = service

    def setPatient(self,patient):
        self.patient = patient
    def getPatient(self):
        return self.patient
    def setDate(self,date):
        self.date = date
    def getDate(self):
        return self.date
    def setDentist(self,dentist):
        self.dentist = dentist
    def getDentist(self):
        return self.dentist
    def setService(self,service):
        self.service = service
    def getService(self):
        return self.service

    def __str__(self):
        return f"Appointment confirmed for patient {self.patient.getPatientName()} on {self.date} with Dr. {self.dentist.getStaffName()} for {self.service.getServiceName()}."
